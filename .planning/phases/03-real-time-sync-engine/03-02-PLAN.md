# Plan 03-02: Change Broadcasting — CRUD Events

<objective>
Wire up real-time broadcasting for all card and list CRUD operations. After a successful DB write, the backend emits a socket event to the board room. The frontend listens for these events and dispatches Redux actions to update state for all other connected users. Implements the "Option A" duplicate prevention strategy: the acting user uses the API response, other users use socket events.
</objective>

<must_haves>
- card:create event broadcasts new card to other users in the room
- card:move event broadcasts card position changes (drag-drop) to other users
- card:update event broadcasts title changes to other users
- card:delete event broadcasts card removal to other users
- list:create event broadcasts new list to other users
- No duplicate updates — acting user uses API response, others use socket
- All events emitted AFTER successful DB write (never before)
- Payloads are minimal (entity data only, not full board)
</must_haves>

<action>
Create `client/src/socket/boardEvents.js`:

```javascript
import { getSocket } from './socketClient';
import { store } from '../store/index';

// Track the current user's pending actions to prevent duplicate updates
let pendingActions = new Set();

export const addPendingAction = (actionKey) => {
  pendingActions.add(actionKey);
  // Auto-cleanup after 5 seconds (safety net)
  setTimeout(() => pendingActions.delete(actionKey), 5000);
};

export const setupBoardListeners = () => {
  const socket = getSocket();
  if (!socket) return;

  // Remove any existing listeners first (prevent duplicates on re-mount)
  socket.off('card:created');
  socket.off('card:updated');
  socket.off('card:deleted');
  socket.off('list:created');

  socket.on('card:created', (data) => {
    const key = `card:create:${data.card.id}`;
    if (pendingActions.has(key)) {
      pendingActions.delete(key);
      return; // This user created it — already handled by API response
    }
    store.dispatch({ type: 'board/socketCardCreated', payload: data });
  });

  socket.on('card:updated', (data) => {
    const key = `card:update:${data.card.id}`;
    if (pendingActions.has(key)) {
      pendingActions.delete(key);
      return;
    }
    store.dispatch({ type: 'board/socketCardUpdated', payload: data });
  });

  socket.on('card:deleted', (data) => {
    const key = `card:delete:${data.cardId}`;
    if (pendingActions.has(key)) {
      pendingActions.delete(key);
      return;
    }
    store.dispatch({ type: 'board/socketCardDeleted', payload: data });
  });

  socket.on('list:created', (data) => {
    const key = `list:create:${data.list.id}`;
    if (pendingActions.has(key)) {
      pendingActions.delete(key);
      return;
    }
    store.dispatch({ type: 'board/socketListCreated', payload: data });
  });
};

export const cleanupBoardListeners = () => {
  const socket = getSocket();
  if (!socket) return;
  socket.off('card:created');
  socket.off('card:updated');
  socket.off('card:deleted');
  socket.off('list:created');
};
```

**Key design decisions:**
- Uses `pendingActions` Set to implement Option A duplicate prevention
- Direct `store.dispatch` with plain action types (handled by slice `extraReducers` or `reducers`)
- `cleanupBoardListeners` removes all listeners (called on unmount)
- Listeners are idempotent — `socket.off` before `socket.on` prevents duplicates
</action>

```javascript
// Add these inside the `reducers: { ... }` block, after moveCardOptimistic:

// Socket-driven reducers (for other users' updates)
socketCardCreated: (state, action) => {
  const { card, listId, boardId } = action.payload;
  const board = state.boards.find(b => b.id === boardId);
  if (board) {
    const list = board.lists?.find(l => l.id === listId);
    if (list) {
      if (!list.cards) list.cards = [];
      // Avoid duplicates
      if (!list.cards.find(c => c.id === card.id)) {
        list.cards.push(card);
        list.cards.sort((a, b) => a.rank.localeCompare(b.rank));
      }
    }
  }
},

socketCardUpdated: (state, action) => {
  const { card, boardId } = action.payload;
  const board = state.boards.find(b => b.id === boardId);
  if (!board) return;

  // Remove card from its current list
  board.lists?.forEach(list => {
    if (list.cards) {
      list.cards = list.cards.filter(c => c.id !== card.id);
    }
  });

  // Add card to its (possibly new) list
  const targetList = board.lists?.find(l => l.id === card.listId);
  if (targetList) {
    if (!targetList.cards) targetList.cards = [];
    targetList.cards.push(card);
    targetList.cards.sort((a, b) => a.rank.localeCompare(b.rank));
  }
},

socketCardDeleted: (state, action) => {
  const { cardId, listId, boardId } = action.payload;
  const board = state.boards.find(b => b.id === boardId);
  if (board) {
    const list = board.lists?.find(l => l.id === listId);
    if (list?.cards) {
      list.cards = list.cards.filter(c => c.id !== cardId);
    }
  }
},

socketListCreated: (state, action) => {
  const { list, boardId } = action.payload;
  const board = state.boards.find(b => b.id === boardId);
  if (board) {
    if (!board.lists) board.lists = [];
    // Avoid duplicates
    if (!board.lists.find(l => l.id === list.id)) {
      board.lists.push(list);
      board.lists.sort((a, b) => a.rank.localeCompare(b.rank));
    }
  }
},
```
