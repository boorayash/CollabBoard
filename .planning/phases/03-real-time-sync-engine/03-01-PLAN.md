<task id="03" title="Create Socket.IO server initialization with CORS and JWT auth">
<read_first>
- server/src/server.js
- server/src/middlewares/authMiddleware.js
- server/.env.example
</read_first>

<action>
Create `server/src/socket/index.js`:

```javascript
const { Server } = require('socket.io');
const jwt = require('jsonwebtoken');
const { setupBoardRooms } = require('./boardRooms');

let io;

const initSocket = (server) => {
  io = new Server(server, {
    cors: {
      origin: '*',   // In production, restrict to frontend URL
      methods: ['GET', 'POST'],
    },
  });

  // Authentication middleware — verify JWT from handshake
  io.use((socket, next) => {
    const token = socket.handshake.auth?.token;
    if (!token) {
      return next(new Error('Authentication error: No token'));
    }
    try {
      const decoded = jwt.verify(token, process.env.JWT_SECRET);
      socket.userId = decoded.id;
      next();
    } catch (err) {
      return next(new Error('Authentication error: Invalid token'));
    }
  });

  io.on('connection', (socket) => {
    console.log(`User connected: ${socket.userId} (socket: ${socket.id})`);
    setupBoardRooms(socket, io);

    socket.on('disconnect', () => {
      console.log(`User disconnected: ${socket.userId} (socket: ${socket.id})`);
    });
  });
};

const getIO = () => {
  if (!io) throw new Error('Socket.IO not initialized');
  return io;
};

module.exports = { initSocket, getIO };
```

**Key design decisions:**
- `getIO()` exported so controllers can call it to emit events after DB writes
- JWT token passed via `socket.handshake.auth.token` (not query params — more secure)
- `socket.userId` set after auth so room handlers know who the user is
</action>

<task id="04" title="Create board room join/leave handlers">
<read_first>

<action>
Create `server/src/socket/boardRooms.js`:

```javascript
const setupBoardRooms = (socket, io) => {
  // Join a board room
  socket.on('board:join', (boardId) => {
    // Leave any previous board room (user can only be in one board at a time)
    const currentRooms = Array.from(socket.rooms);
    currentRooms.forEach((room) => {
      if (room !== socket.id && room.startsWith('board:')) {
        socket.leave(room);
        console.log(`User ${socket.userId} left room ${room}`);
      }
    });

    const roomName = `board:${boardId}`;
    socket.join(roomName);
    console.log(`User ${socket.userId} joined room ${roomName}`);
  });

  // Leave a board room explicitly
  socket.on('board:leave', (boardId) => {
    const roomName = `board:${boardId}`;
    socket.leave(roomName);
    console.log(`User ${socket.userId} left room ${roomName}`);
  });
};

module.exports = { setupBoardRooms };
```

**Key:** Room naming convention is `board:{boardId}`. Users auto-leave previous board rooms when joining a new one (single-board focus model).
</action>

<action>
Create `client/src/socket/socketClient.js`:

```javascript
import { io } from 'socket.io-client';

let socket = null;

export const connectSocket = (token) => {
  if (socket?.connected) return socket;

  const API_URL = import.meta.env.VITE_API_URL || 'http://localhost:5000';
  // Socket connects to the root server, not the /api path
  const baseURL = API_URL.replace('/api', '');

  socket = io(baseURL, {
    auth: { token },
    reconnection: true,
    reconnectionDelay: 1000,
    reconnectionAttempts: 5,
  });

  socket.on('connect', () => {
    console.log('Socket connected:', socket.id);
  });

  socket.on('connect_error', (err) => {
    console.error('Socket connection error:', err.message);
  });

  socket.on('disconnect', (reason) => {
    console.log('Socket disconnected:', reason);
  });

  return socket;
};

export const getSocket = () => socket;

export const disconnectSocket = () => {
  if (socket) {
    socket.disconnect();
    socket = null;
  }
};

export const joinBoard = (boardId) => {
  if (socket?.connected) {
    socket.emit('board:join', boardId);
  }
};

export const leaveBoard = (boardId) => {
  if (socket?.connected) {
    socket.emit('board:leave', boardId);
  }
};
```

**Key decisions:**
- Singleton pattern — only one socket connection per session
- `connectSocket(token)` called after login
- `joinBoard`/`leaveBoard` helpers abstract room management
- Connection URL strips `/api` suffix since Socket.IO connects to root
</action>
